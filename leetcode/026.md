class Solution {
    public int removeDuplicates(int[] nums) {
       
    //快慢指针
    int slow = 0;
    for (int quick = 1; quick < nums.length; quick++) {
      if (nums[slow]==nums[quick]){
        //让quick自动加一就好了
      }else {
        //slow移动一位，让slow位=quick位
        slow++;
        nums[slow]=nums[quick];
      }
    }
    return ++slow;
  } 
    
}
