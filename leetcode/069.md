    }
    //二分法
    public static int mySqrt(int x) {
        long left=0,right=x/2+1;
        long square=0,mid=0;
        while(left<right){
            // try{
            //     Thread.sleep(1000);
            // }catch(Exception e){
            // }
            mid=(left+right+1)>>1;
            square=mid*mid;
            System.out.println("square:"+square);
            if(square>x){
                right=mid-1;
            }else{
                left=mid;
            }
        }        
        return (int)left;
    }
}
