class Solution {
    public int removeElement(int[] nums, int val) {
        int slow=0;
        for(int quick=0;quick<nums.length;quick++){
            if(nums[quick]==val){
                //快指针自加
            }else{
                nums[slow]=nums[quick];
                slow++;
            }
        }
        return slow;
    }

}
